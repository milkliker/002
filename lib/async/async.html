一、使用 data-async="/url/to/get/data"

如果在 HTMT Attribute 中指定 data-async，配合事件处理可以行到想要的内容。

点我

二、纯 js 调用

支持事件传数据、定义处理事件等，具体看这个参数描述：

/* 以 js 方式调用 async 组件
 * @param options {Object} 传入的对象，支持三个属性/方法
 *  {
 *     url: {String} [可选] 请求的地址，默认为 $(this).data('async')
 *     eventType: [可选] {String} 触发异步请求的事件，默认为 click
 *     callback: [可选] {Function} 接入一个参数，当参数非空是为请求成功，否则为失败
 *     data: {Object/Function} [可选] 需要发送的数据，当 data 为 Function 的时候，执行它
 *     method: {String} [可选] 请求数据的式，GET / POST，默认为 GET
 *  }
 */

使用方法: 

// 纯 js 调用
var callback = function(data){
		console.log(data);
	},
	options = {
		url: '/ajax/tag/autocomplete/?category=root&cityEnglishName=shanghai&key=&term=nu',
		callback: callback,
		eventType: 'blur'
	}

$('#hello').async(options);

三、支持事件监听

  • fetching: 开始请求数据 
    $('[data-async]').on('async_before', function(){
	console.log('你点我啦，等一等，数据马上来');
});
  • done: 完成数据获取 
    // 回调函数第二个参数是请求返回的数据
$('[data-async]').on('async_success', function(e, data){
	console.log('数据是: ', data);
});
  • error: 获取资源时出错 
    $('[data-async]').on('async_error', function(){
	console.log('出错啦');
});
